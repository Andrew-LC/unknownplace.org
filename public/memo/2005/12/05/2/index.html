#!/usr/bin/perl

use strict;
use warnings;

use LWP::UserAgent;

my ($newmail) = qx{ find $ENV{HOME}/Maildir/new -type f | wc -l } =~ /(\d+)/;
my ($newbloglines) = LWP::UserAgent->new
    ->get('http://rpc.bloglines.com/update?user=Bloglines登録メアド&ver=1')
    ->content =~ /(\d+)/;

print "Mail($newmail) Feed($newbloglines) |\n";
