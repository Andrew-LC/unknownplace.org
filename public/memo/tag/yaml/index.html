use strict; use warnings;

use Data::Dumper;
use YAML;

my $config = {
    default => [ 1, { '*' => 1 }, ],
    deny    => ['script'],
};

my $yaml = Dump($config);

print $yaml;
print Dumper( Load($yaml) );

  - '*': 1

   Line: 5
   Document: 1

This code works fine with YAML::Syck.
