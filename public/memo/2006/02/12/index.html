sub fuzzyfind { my ( $self, $id ) = @_;

    my ( $exist, $uuid );
    $exist = $self->find( { uuid => $id } ) and $uuid++
        if $id and $id =~ /\w/;

    defined $exist ? ( $exist, $uuid ) : $self->find($id);
}
