ダサい解決法

package MyProvider;
use base qw/Template::Provider/;

sub _load {
    my ($data, $error) = shift->SUPER::_load(@_);

    if ($data && $data->{text}) {
        $data->{text} =~ s/(\[%[ ]+INCLUDE[ ]+["'])(.+)(["'].*?%\])/$1common:$2$3/;
    }

    ($data, $error);
}
